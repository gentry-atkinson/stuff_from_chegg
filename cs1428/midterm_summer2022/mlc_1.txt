
Write nested for loops (only the loops, not a full program) which will print this as their output:

4 3 2 1

3 2 1

2 1

1

#include <iostream>

using namespace std;

int main(){
   
   // looping i from 1 to 5
   for(int i = 1;i<=5;i++){
      // looping j from i to 1
      for(int j = i;j>=1;j--){
         // printing value of j
         cout<<j<<" ";
      }
      // printing new line
      cout<<endl;
   }
                 
   return 0;
}
